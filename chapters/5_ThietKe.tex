        \textbf{Use case ID} & UC--SA--PAY--02 \\

        \hline
        \textbf{Actor} & Người quản trị hệ thống (Super Admin) \\
        \hline
        \textbf{Description} & Tính và thực hiện phân chia doanh thu giữa hệ thống và chủ trạm theo kỳ (ngày/tuần/tháng).\\
        \hline
        \textbf{Preconditions} & 
        \begin{itemize}
            \item Super Admin đăng nhập thành công vào hệ thống.
            \item Dữ liệu giao dịch đã “hoàn tất” trong kỳ chọn, đã đối soát cơ bản.
            \item Cấu hình tỷ lệ chia, phí nền tảng, ngưỡng chi trả tối thiểu… đã thiết lập.
        \end{itemize}\\
        \hline
        \textbf{Postconditions} & 
        \begin{itemize}
            \item Hệ thống lưu version cấu hình phân chia doanh thu
            \item Lịch sử cấu hình được cập nhật
        \end{itemize}\\
        \hline
        \textbf{Normal flows} & 
        \begin{enumerate}[label=\arabic*.]
            \item Super Admin chọn chức năng Phân chia doanh thu.
            \item Nhập tỷ lệ chia.
            \item Xác nhận lưu cấu hình.
            \item Hệ thống cập nhật tỷ lệ chia mới.
        \end{enumerate}\\
        \hline
        \textbf{Alternative flows} & 
        \begin{itemize}
            \item 2a. Super Admin chọn áp dụng tỷ lệ chia riêng cho từng trạm thay vì áp dụng chung.
            \item 2b. Super Admin xem lại lịch sử cấu hình trước đó và rollback.
        \end{itemize}\\
        \hline
        \textbf{Exception} & 
        \begin{itemize}
            \item 3a. Nhập sai định dạng hoặc tỷ lệ không hợp lệ: Hệ thống báo lỗi.
            \item 4a. Lưu thất bại (lỗi DB/hệ thống): Thông báo lỗi, không thay đổi cấu hình cũ.
        \end{itemize}\\
        \hline
        \caption{Use Case Phân chia doanh thu}
        \label{tab:uc_sa_pay_02}
    \end{longtable}

        \end{enumerate}\\

    \item Chọn chức năng: Người dùng chọn menu “Phân chia doanh thu”.
    \item Tiếp theo, Super Admin chọn phạm vi áp dụng cho việc phân chia này (có thể là cho toàn hệ thống hoặc theo từng trạm).

    \begin{itemize}
        \item Nếu CÓ (Yes):

    \end{itemize} 
    \item Nhập cấu hình mới: (Đây là điểm hợp nhất của luồng "Không xem lịch sử" và "Không rollback").
    \begin{itemize}
        \item Super Admin thực hiện nhập tỷ lệ phân chia mới.
        \item Super Admin xác nhận lưu cấu hình.
    \end{itemize}
    \item Quyết định (Xác thực): Hệ thống sẽ kiểm tra xem định dạng, tỷ lệ có hợp lệ không?
    \begin{itemize}
        \item Nếu KHÔNG (No):
        \begin{itemize}
            \item Hệ thống yêu cầu điền lại thông tin.
            \item Quy trình lặp lại, quay về bước nhập tỷ lệ phân chia để Super Admin sửa lại.
        \end{itemize}
        \item Nếu CÓ (Yes):
        \begin{itemize}
            \item Hệ thống thực hiện lưu cấu hình mới.
            \item Quy trình kết thúc.
        \end{itemize}
    \end{itemize}
\end{enumerate}

\noindent \textbf{Luồng chính của biểu đồ trình tự:}
\begin{enumerate}[font=\bfseries]
    \item Người dùng mở màn hình tìm kiếm.
    \item Ứng dụng xin quyền và lấy vị trí hiện tại từ GPS hoặc vị trí được nhập thủ công.
    \item Ứng dụng gửi vị trí (và bộ lọc nếu có) lên máy chủ.
    \item Máy chủ xử lý, lọc và trả về một danh sách các trạm sạc.
    \item Ứng dụng yêu cầu MapSDK vẽ các điểm đánh dấu (marker) của trạm lên bản đồ.
    \item Ứng dụng hiển thị danh sách trạm trên bản đồ cho người dùng.
    \item Khi người dùng chọn một trạm, ứng dụng gọi máy chủ để lấy thông tin chi tiết của trạm đó và hiển thị ra màn hình.
\end{enumerate}
\textbf{Các luồng phụ:}
\begin{enumerate}[font=\bfseries]
    \item Từ chối quyền / GPS tắt: Nếu GPS không cấp quyền hoặc bị lỗi, ứng dụng sẽ yêu cầu người dùng nhập vị trí thủ công.
    \item Không tìm thấy trạm: Nếu máy chủ trả về danh sách rỗng, ứng dụng sẽ hiển thị thông báo "Không tìm thấy trạm".
    \item Điều hướng (Tùy chọn): Nếu người dùng nhấn nút chỉ đường, ứng dụng sẽ yêu cầu MapSDK mở một ứng dụng bản đồ bên ngoài.
    \item Lỗi mạng (Tùy chọn): Nếu quá trình gọi máy chủ bị lỗi, ứng dụng sẽ hiển thị thông báo lỗi.
\end{enumerate}
