    \begin{center}
    \renewcommand{\arraystretch}{1.5}
        \begin{longtable}{|m{0.20\textwidth}|m{0.65\textwidth}|}
        \hline
        \textbf{Use case ID} & UC--SA--PAY--02 \\
        \hline
        \textbf{Use case name} & Phân chia doanh thu \\
        \hline
        \textbf{Actor} & Người quản trị hệ thống (Super Admin) \\
        \hline
        \textbf{Description} & Tính và thực hiện phân chia doanh thu giữa hệ thống và chủ trạm theo kỳ (ngày/tuần/tháng).\\
        \hline
        \textbf{Preconditions} & 
        \begin{itemize}
            \item Super Admin đăng nhập thành công vào hệ thống.
            \item Dữ liệu giao dịch đã “hoàn tất” trong kỳ chọn, đã đối soát cơ bản.
            \item Cấu hình tỷ lệ chia, phí nền tảng, ngưỡng chi trả tối thiểu… đã thiết lập.
        \end{itemize}\\
        \hline
        \textbf{Postconditions} & 
        \begin{itemize}
            \item Hệ thống lưu version cấu hình phân chia doanh thu
            \item Lịch sử cấu hình được cập nhật
        \end{itemize}\\
        \hline
        \textbf{Normal flows} & 
        \begin{enumerate}[label=\arabic*.]
            \item Super Admin chọn chức năng Phân chia doanh thu.
            \item Nhập tỷ lệ chia.
            \item Xác nhận lưu cấu hình.
            \item Hệ thống cập nhật tỷ lệ chia mới.
        \end{enumerate}\\
        \hline
        \textbf{Alternative flows} & 
        \begin{itemize}
            \item 2a. Super Admin chọn áp dụng tỷ lệ chia riêng cho từng trạm thay vì áp dụng chung.
            \item 2b. Super Admin xem lại lịch sử cấu hình trước đó và rollback.
        \end{itemize}\\
        \hline
        \textbf{Exception} & 
        \begin{itemize}
            \item 3a. Nhập sai định dạng hoặc tỷ lệ không hợp lệ: Hệ thống báo lỗi.
            \item 4a. Lưu thất bại (lỗi DB/hệ thống): Thông báo lỗi, không thay đổi cấu hình cũ.
        \end{itemize}\\
        \hline
        \caption{Use Case Phân chia doanh thu}
        \label{tab:uc_sa_pay_02}
    \end{longtable}

            \item Hệ thống hiển thị kết quả; khi Super Admin chọn một giao dịch, hệ thống mở chi tiết thông tin giao dịch.

        \end{enumerate}\\
