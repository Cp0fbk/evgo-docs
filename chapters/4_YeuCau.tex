\subsubsubsection{Chức năng chung của hệ thống}
\begin{enumerate}[label=\textbf{\alph*)}]
 \item \textbf{Đăng nhập, đăng xuất}
 \begin{itemize}
    \item Người dùng có thể đăng nhập vào hệ thống bằng email/số điện thoại.
    \item Sau khi đăng nhập thành công, người dùng có quyền truy cập các chức năng theo vai trò.
    \item Cho phép đăng xuất để kết thúc phiên làm việc, đảm bảo an toàn bảo mật thông tin.
 \end{itemize}
 \item \textbf{Quản lý tài khoản cá nhân}
 \begin{itemize}
    \item Người dùng có thể chỉnh sửa thông tin cá nhân.
    \item Người dùng có thể thay đổi/khôi phục mật khẩu thông qua email/số điện thoại đã đăng ký.
 \end{itemize}
\end{enumerate}
\subsubsubsection{Chức năng của Người quản trị hệ thống}
Với vai trò là cấp quản trị cao nhất, Người quản lý hệ thống (Super Admin) chịu trách nhiệm quản lý toàn bộ hệ thống, bảo đảm tính minh bạch, ổn định và hiệu quả trong vận hành. Các yêu cầu chức năng cụ thể bao gồm:
\begin{enumerate}[label=\textbf{\alph*)}]
 \item \textbf{Quản lý tài khoản}
 \begin{itemize}
    \item Người quản lý hệ thống có khả năng truy cập và theo dõi danh sách toàn bộ người dùng đã đăng ký trong hệ thống.
    \item Trong trường hợp phát hiện hành vi vi phạm hoặc sử dụng sai mục đích, Người quản lý hệ thống được quyền khóa tạm thời hoặc mở khóa lại tài khoản để đảm bảo an toàn cho hệ thống.
    \item Người quản lý hệ thống xem và quản lý danh sách các chủ trạm đã đăng ký, từ đó nắm bắt tình hình hoạt động tổng thể. Người quản lý hệ thống gửi mail kích hoạt tài khoản đến chủ trạm.
 \end{itemize}
 \item \textbf{Quản lý trạm sạc}
 \begin{itemize}
    \item Người quản lý hệ thống theo dõi tình trạng hoạt động của tất cả trạm sạc trên hệ thống theo thời gian thực, bao gồm thông tin về số lượng cổng sạc, trạng thái và mức độ sẵn sàng.
 \end{itemize}
 \item \textbf{Báo cáo và thống kê tổng hợp}
 \begin{itemize}
    \item Người quản lý hệ thống được cung cấp công cụ báo cáo thống kê toàn diện, bao gồm số lượng người dùng, trạm sạc, lượt đặt chỗ và doanh thu toàn hệ thống.
    \item Các dữ liệu này có thể được biểu diễn dưới dạng biểu đồ, phân tích theo khu vực hoặc theo loại phương tiện, giúp hỗ trợ việc đánh giá tình hình và hoạch định chiến lược phát triển trong tương lai.
 \end{itemize}  
 \item \textbf{Giám sát giao dịch và doanh thu}
 \begin{itemize}
    \item Hệ thống cung cấp cho Người quản lý hệ thống khả năng theo dõi toàn bộ luồng giao dịch giữa người dùng và chủ trạm, bảo đảm sự minh bạch và chính xác trong quá trình thanh toán.
    \item Trong trường hợp hệ thống áp dụng mô hình thu phí, Người quản lý hệ thống có trách nhiệm giám sát và thực hiện phân chia doanh thu giữa các bên liên quan.
 \end{itemize}
\end{enumerate}

\subsubsubsection{Chức năng của Nhân viên hỗ trợ khách hàng}
Customer Service Staff hay còn gọi là nhân viên chăm sóc khách hàng tiếp nhận và giải quyết khiếu nại, hỗ trợ người dùng và chủ trạm.
\begin{enumerate}[label=\textbf{\alph*)}]
 \item \textbf{Quản lý trạm sạc toàn hệ thống}
 \begin{itemize}
    \item CSS theo dõi tình trạng hoạt động của tất cả trạm sạc trên hệ thống theo thời gian thực, bao gồm thông tin về số lượng cổng sạc, trạng thái và mức độ sẵn sàng.
 \end{itemize}
 \item \textbf{Giao tiếp với người dùng và chủ trạm}
 \begin{itemize}
    \item CSS tiếp nhận và quản lý khiếu nại phát sinh từ người dùng hoặc chủ trạm.
   \item Trong trường hợp phát hiện hành vi vi phạm hoặc sử dụng sai mục đích, CSS được quyền khóa tạm thời hoặc mở khóa lại tài khoản để đảm an toàn cho hệ thống.
 \end{itemize}
 \end{enumerate}

\subsubsection{Yêu cầu phi chức năng}
\subsubsubsection{Hiệu năng (Performance)}
Yêu cầu về tốc độ và khả năng phản hồi của hệ thống.
\begin{itemize}
    \item \textbf{Thời gian phản hồi (Response Time):}
    \begin{itemize}
        \item Các thao tác cơ bản (đăng nhập, xem danh sách, xem chi tiết trạm) phải phản hồi dưới 2 giây.
        \item Kết quả tìm kiếm và lọc trạm sạc phải hiển thị dưới 3 giây
        \item Quy trình thanh toán phải xử lý trong khoảng 5 giây tùy thuộc vào cổng thanh toán bên thứ 3.
    \end{itemize}
    \item \textbf{Xử lý thời gian thực (Real-time Processing):}
    \begin{itemize}
        \item Trạng thái cổng sạc (Trống/Đang sạc) và quá trình sạc (mô phỏng phần trăm pin) phải được cập nhật trên ứng dụng người dùng và dashboard chủ trạm với độ trễ thấp.
    \end{itemize}
    \item \textbf{Khả năng chịu tải (Concurrency):}
    \begin{itemize}
        \item Hệ thống phải hỗ trợ tối thiểu 100 -- 500 người dùng truy cập đồng thời mà không bị sập.
        \item Hỗ trợ xử lý đồng thời 20 -- 50 giao dịch đặt chỗ/thanh toán cùng một lúc.
    \end{itemize}
\end{itemize}

\subsubsubsection{Độ tin cậy \& Sẵn sàng (Reliability \& Availability)}
Đảm bảo hệ thống luôn hoạt động ổn định.
\begin{itemize}
    \item \textbf{Thời gian hoạt động (Uptime):} Hệ thống phải đảm bảo hoạt động 99.9\% thời gian vì liên quan đến việc di chuyển và sạc điện của người dùng.
    \item \textbf{Sao lưu dữ liệu (Data Backup):} Dữ liệu về giao dịch, lịch sử sạc và tài khoản người dùng phải được tự động sao lưu hàng ngày và có cơ chế khôi phục nhanh chóng khi có sự cố.
    \item \textbf{Xử lý lỗi (Fault Tolerance):} Nếu module thanh toán gặp lỗi, hệ thống vẫn phải cho phép người dùng xem bản đồ hoặc tra cứu thông tin trạm, không crash toàn bộ ứng dụng.
\end{itemize}

\subsubsubsection{Bảo mật (Security)}
Đảm bảo an toàn dữ liệu người dùng, chủ trạm và hệ thống.
\begin{itemize}
    \item \textbf{Mã hóa dữ liệu:}
    \begin{itemize}
        \item Mật khẩu người dùng phải được mã hóa trong cơ sở dữ liệu.
        \item Mọi truyền tải dữ liệu qua mạng phải sử dụng giao thức an toàn (HTTPS/SSL).
    \end{itemize}
    \item \textbf{Xác thực \& Phân quyền (Authentication \& Authorization):}
    \begin{itemize}
        \item Cơ chế xác thực an toàn cho việc đăng nhập đa nền tảng.
        \item Phân quyền chặt chẽ (RBAC - Role Based Access Control): Người dùng thường không được truy cập API của Admin; Chủ trạm A không được xem doanh thu của Chủ trạm B.
    \end{itemize}
    \item \textbf{Bảo mật thanh toán:} Không lưu trực tiếp thông tin thẻ ngân hàng/CVV của người dùng trên hệ thống. Phải tích hợp qua các cổng thanh toán uy tín (Momo, VNPay,...) và tuân thủ chuẩn PCI DSS.
\end{itemize}

\subsubsubsection{Khả năng sử dụng (Usability)}
Đảm bảo trải nghiệm người dùng (UX/UI).
\begin{itemize}
    \item \textbf{Giao diện người dùng (User Interface):}
    \begin{itemize}
        \item \textit{Với User:} Thiết kế Mobile-first, thao tác đặt lịch và tìm đường phải tối giản. Giao diện trực quan, icon rõ ràng.
        \item \textit{Với Super Admin/Station Owner:} Dashboard quản lý phải hiển thị tốt trên trình duyệt web (Desktop), biểu đồ thống kê dễ đọc.
    \end{itemize}
    \item \textbf{Ngôn ngữ:} Hỗ trợ Tiếng Anh để có thể tiếp cận người dùng từ bất kỳ quốc gia nào.
    \item \textbf{Thông báo:} Thông báo lỗi phải rõ ràng, dễ hiểu.
\end{itemize}

\subsubsubsection{Khả năng mở rộng (Scalability)}
Chuẩn bị cho việc phát triển trong tương lai.
\begin{itemize}
    \item \textbf{Mở rộng cơ sở dữ liệu:} Hệ thống phải thiết kế để có thể lưu trữ lượng lớn dữ liệu lịch sử giao dịch trong tương lai.
    \item \textbf{Kiến trúc:} Thiết kế theo hướng Modular Monolith để dễ dàng thêm các module mới mà không ảnh hưởng đến toàn bộ hệ thống.
\end{itemize}

\subsubsubsection{Khả năng tương thích \& Tích hợp (Interoperability \& Integration)}
Khả năng giao tiếp với bên thứ 3 và phần cứng.
\begin{itemize}
    \item \textbf{Bản đồ:} Tích hợp tốt với Google Maps API hoặc Mapbox để định vị, hiển thị trạm và điều hướng chính xác.
    \item \textbf{IoT \& Chuẩn sạc:}
    \begin{itemize}
        \item Dù hiện tại là mô phỏng, nhưng kiến trúc hệ thống nên hướng tới việc hỗ trợ giao thức \textbf{OCPP (Open Charge Point Protocol)}. Đây là chuẩn quốc tế để phần mềm quản lý giao tiếp với các trụ sạc thực tế.
    \end{itemize}
    \item \textbf{Cổng thanh toán:} API phải thiết kế mở để dễ dàng tích hợp thêm các ví điện tử mới trong tương lai.
\end{itemize}

\subsubsubsection{Yêu cầu về môi trường \& Triển khai}
Ứng dụng cần hoạt động ổn định trên các nền tảng và thiết bị phổ biến.
    \begin{itemize}
        \item User: Ứng dụng chạy mượt trên Android (từ version 10 trở lên) và iOS (từ version 14 trở lên).
        \item Super Admin/Station Owner/Customer Service Staff: Chạy tốt trên các trình duyệt phổ biến (Chrome, Firefox, Safari, Edge).
    \end{itemize}
